58  	String getCentury() {
59  		String result = null;
60  		int individnummerInt = Integer.parseInt(getIndividnummer());
61  		int birthYear = Integer.parseInt(get2DigitBirthYear());
62  		if (individnummerInt <= 499) {
63  			result = "19";
64  		} else if (individnummerInt >= 500 && birthYear < 40) {
65  			result = "20";
66  		} else if (individnummerInt >= 500 && individnummerInt <= 749 && birthYear > 54) {
67  			result = "18";
68  		} else if (individnummerInt >= 900 && birthYear > 39) {
69  			result = "19";
70  		}
71  		return result;
72  	}
