63  	private static List<Navn> genererNavn(final int antall, final KJONN kjonn) {
64  		List<Navn> navneliste = new ArrayList<Navn>(antall);
65  		KJONN kjonnSwitch = kjonn;
66  		while (navneliste.size() < antall) {
67  			if (KJONN.erBegge(kjonn)) {
68  				kjonnSwitch = KJONN.byttKjonn(kjonn);
69  			}
70  			Navn navn = genererNavn(kjonnSwitch);
71  			if (!navneliste.contains(navn)) {
72  				navneliste.add(navn);
73  			}
74  		}
75  		return navneliste;
76  	}

78  	private static Navn genererNavn(final KJONN kjonn) {
79  		String fnavn, mnavn = null, enavn;
80  		int indexF = 0;
81  		if (KJONN.erKvinne(kjonn)) {
82  			indexF = new Random().nextInt(kvinnenavn.size() - 1);
83  			fnavn = kvinnenavn.get(indexF);
84  		} else {
85  			indexF = new Random().nextInt(mannsnavn.size() - 1);
86  			fnavn = mannsnavn.get(indexF);
87  		}
88  		if (genererMellomnavn(kjonn)) {
89  			int indexM = new Random().nextInt(etternavn.size() - 1);
90  			mnavn = etternavn.get(indexM);
91  		}
92  		int indexE = new Random().nextInt(etternavn.size() - 1);
93  		enavn = etternavn.get(indexE);
94  		return new Navn(fnavn, mnavn, enavn);
95  	}

97  	private static boolean genererMellomnavn(KJONN kjonn) {
98  		if (KJONN.erKvinne(kjonn)) {
99  			if (new Random().nextInt(100) <= caAntallKvinnerSomHarMellomnavnIProsent) {
100 				return true;
101 			}
102 		} else {
103 			if (new Random().nextInt(100) <= caAntallMennSomHarMellomnavnIProsent) {
104 				return true;
105 			}
106 		}
107 		return false;
108 	}

110 	private static List<String> csv2List(InputStream is) {
111 		InputStreamReader isr = new InputStreamReader(is, Charset.forName("UTF-8"));
112 		BufferedReader br = new BufferedReader(isr);
113 		List<String> vList = new ArrayList<String>();
114 		String[] array;
115 		String line;
116 		try {
117 			while ((line = br.readLine()) != null) {
118 				array = line.split("[ ]*,[ ]*");
119 				for (int i = 0; i < array.length; i++) {
120 					vList.add(array[i]);
121 				}
122 			}
123 			br.close();

125 		} catch (IOException e) {

127 		}
