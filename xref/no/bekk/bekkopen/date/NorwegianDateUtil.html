22  	/***
23  	 * Adds the given number of working days to the given date. A working day is
24  	 * specified as a regular Norwegian working day, excluding weekends and all
25  	 * national holidays.
26  	 * <p/>
27  	 * Example 1:<br/>
28  	 * - Add 5 working days to Wednesday 21.03.2007 -> Yields Wednesday
29  	 * 28.03.2007. (skipping saturday and sunday)<br/>
30  	 * <p/>
31  	 * Example 2:<br/>
32  	 * - Add 5 working days to Wednesday 04.04.2007 (day before
33  	 * easter-long-weekend) -> yields Monday 16.04.2007 (skipping 2 weekends and
34  	 * 3 weekday holidays).
35  	 * 
36  	 * @param date
37  	 *            The original date.
38  	 * @param days
39  	 *            The number of working days to add.
40  	 * @return The new date.
41  	 */
42  	public static Date addWorkingDaysToDate(Date date, int days) {
43  		Calendar cal = dateToCalendar(date);
44  
45  		for (int i = 0; i < days; i++) {
46  			cal.add(Calendar.DATE, 1);
47  			while (!isWorkingDay(cal)) {
48  				cal.add(Calendar.DATE, 1);
49  			}
50  		}
51  
52  		return cal.getTime();
53  	}

92  	/***
93  	 * Get a set of holidays for a given year.
94  	 * 
95  	 * @param year
96  	 *            The year to get holidays for.
97  	 * @return The set of dates.
98  	 */
99  	private static Set<Date> getHolidaySet(int year) {
100 		Integer yInt = Integer.valueOf(year);
101 		if (holidays == null) {
102 			holidays = new HashMap<Integer, Set<Date>>();
103 		}
104 		if (!holidays.containsKey(yInt)) {
105 			Set<Date> yearSet = new HashSet<Date>();
106 
107 			// Add set holidays.
108 			yearSet.add(getDate(1, Calendar.JANUARY, year));
109 			yearSet.add(getDate(1, Calendar.MAY, year));
110 			yearSet.add(getDate(17, Calendar.MAY, year));
111 			yearSet.add(getDate(25, Calendar.DECEMBER, year));
112 			yearSet.add(getDate(26, Calendar.DECEMBER, year));
113 
114 			// Add movable holidays - based on easter day.
115 			Calendar easterDay = dateToCalendar(getEasterDay(year));
116 
117 			// Sunday before easter.
118 			yearSet.add(rollGetDate(easterDay, -7));
119 
120 			// Thurday before easter.
121 			yearSet.add(rollGetDate(easterDay, -3));
122 
123 			// Friday before easter.
124 			yearSet.add(rollGetDate(easterDay, -2));
125 
126 			// Easter day.
127 			yearSet.add(easterDay.getTime());
128 
129 			// Second easter day.
130 			yearSet.add(rollGetDate(easterDay, 1));
131 
132 			// "Kristi himmelfart" day.
133 			yearSet.add(rollGetDate(easterDay, 39));
134 
135 			// "Pinse" day.
136 			yearSet.add(rollGetDate(easterDay, 49));
137 
138 			// Second "Pinse" day.
139 			yearSet.add(rollGetDate(easterDay, 50));
140 
141 			holidays.put(yInt, yearSet);
142 		}
143 		return holidays.get(yInt);
144 	}

156 				&& !isHoliday(cal);

160 	 * Check if given Calendar object represents a holiday.
161 	 * 
162 	 * @param cal
163 	 *            The Calendar to check.
164 	 * @return true if holiday, false otherwise.
165 	 */
166 	private static boolean isHoliday(Calendar cal) {
167 		int year = cal.get(Calendar.YEAR);
168 		Set<?> yearSet = getHolidaySet(year);
169 		for (Iterator<?> iterator = yearSet.iterator(); iterator.hasNext();) {
170 			Date date = (Date) iterator.next();
171 			if (checkDate(cal, dateToCalendar(date))) {
172 				return true;
173 			}
174 		}
175 		return false;
176 	}

178 	/***
179 	 * Calculates easter day (sunday) by using Spencer Jones formula found here:
180 	 * <a href="http://no.wikipedia.org/wiki/P%C3%A5skeformelen">Wikipedia -
181 	 * PÃ¥skeformelen</a>
182 	 * 
183 	 * @param year
184 	 *            The year to calculate from.
185 	 * @return The Calendar object representing easter day for the given year.
186 	 */
187 	private static Date getEasterDay(int year) {
188 		int a = year % 19;
189 		int b = year / 100;
190 		int c = year % 100;
191 		int d = b / 4;
192 		int e = b % 4;
193 		int f = (b + 8) / 25;
194 		int g = (b - f + 1) / 3;
195 		int h = ((19 * a) + b - d - g + 15) % 30;
196 		int i = c / 4;
197 		int k = c % 4;
198 		int l = (32 + (2 * e) + (2 * i) - h - k) % 7;
199 		int m = (a + (11 * h) + (22 * l)) / 451;
200 		int n = (h + l - (7 * m) + 114) / 31; // This is the month number.
201 		int p = (h + l - (7 * m) + 114) % 31; // This is the date minus one.
202 
203 		Calendar cal = Calendar.getInstance();
204 		cal.set(Calendar.YEAR, year);
205 		cal.set(Calendar.MONTH, n - 1);
206 		cal.set(Calendar.DATE, p + 1);
207 
208 		return cal.getTime();
209 	}
