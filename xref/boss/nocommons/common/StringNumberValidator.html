51  	private static int calculateChecksum(int[] weights, StringNumber number, boolean tverrsum) {
52  		int checkSum = 0;
53          for (int i = 0; i < weights.length; i++) {
54          	int product = weights[i] * number.getAt(weights.length-1-i);
55          	if (tverrsum) {
56                  checkSum += (product > 9? product - 9 : product);
57          	} else {
58                  checkSum += product;
59          	}
60          }
61  		return checkSum;
62  	}

80          }
